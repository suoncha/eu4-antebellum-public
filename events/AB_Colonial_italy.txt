#Eastern America
country_event = {
	id = colonial_italy.1
	title = "colonial_italy.1.t"
	desc = "colonial_italy.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_eastern_america
			is_core = ROOT
		}
		NOT = { exists = IT1 }
		NOT = {
			has_global_flag = NO_CNS
		}
		OR = {
			is_colonial_nation_of = ITA
			is_colonial_nation_of = ITT
			is_colonial_nation_of = ITR
		}
	}
	option = {
		name = "colonial_italy.a"
		change_tag = IT1
		swap_free_idea_group = yes
		swap_non_generic_missions = yes
		ab_update_colonial_eastern_america_color = yes
	}
}
