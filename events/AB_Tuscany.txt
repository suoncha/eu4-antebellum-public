#The Theft of the Mona Lisa
country_event = {
	id = ab_tuscany.3
	title = ab_tuscany.3.t
	desc = ab_tuscany.3.d
	picture = ARTWORK_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		FROM = {
			has_province_modifier = tus_mona_lisa
			NOT = { owned_by = ROOT }
			NOT = {
				owner = {
					OR = {
						is_in_war = {
							defenders = PREV
							defenders = ROOT
						}
						is_in_war = {
							attackers = PREV
							attackers = ROOT
						}
					}
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			FROM = {
				save_global_event_target_as = mona_lia_province
				remove_province_modifier = tus_mona_lisa
			}
		}
	}
	
	option = {
		name = ab_tuscany.3.a
		add_prestige = 10
		hidden_effect = {
			FROM = {
				owner = {
					country_event = {
						id = ab_tuscany.4
					}
				}
			}
		}
		custom_tooltip = AB_STOLEN_MONA_LISA
		capital_scope = {
			add_permanent_province_modifier = {
				name = tus_mona_lisa
				duration = -1
			}
		}
	}
}
