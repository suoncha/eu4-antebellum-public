	is_triggered_only = yes
	
	option = {
		name = ab_lithuania.2.a
		custom_tooltip = LIT_PP_PERMA
		hidden_effect = {
			add_power_projection = {
				type = LIT_perma_pp
				amount = 25
			}
		}
		override_country_name = BLTSC
		set_government_rank = 3
		custom_tooltip = LIT_CULTURE
		set_country_flag = lit_negate_cultures

		# Set the modifier
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						culture_group = scandinavian
						culture_group = Finnic
					}
					has_owner_accepted_culture = no
				}
				add_province_modifier = {
					name = lit_culture_modifier
					duration = -1
				}
			}
		}
	}
}

		complete_mission = lit_marshes

			name = lit_draining

			every_owned_province = {

					has_terrain = marsh
				}

					name = lit_marsh_negate

				}
			}
		}
		define_advisor = {
			type = architect
			skill = 2
			religion = ROOT
			culture = ROOT
			cost_multiplier = 0.5
		}
	}
	
	option = {
		name = ab_lithuania.7.a
	}
}
