#Brazil
country_event = {
	id = colonial_andalusia.1
	title = "colonial_andalusia.1.t"
	desc = "colonial_andalusia.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_brazil
			is_core = ROOT
		}
		NOT = { exists = AD1 }
		NOT = {
			has_global_flag = NO_CNS
		}
		OR = {
			is_colonial_nation_of = ADU
		}
	}
	option = {
		name = "colonial_andalusia.a"
	
		change_tag = AD1
		swap_non_generic_missions = yes
	
	}
}
