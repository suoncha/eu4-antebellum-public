namespace = ab_wendia

#The Destruction of Aachen
country_event = {
	id = ab_wendia.1
	title = ab_wendia.1.t
	desc = ab_wendia.1.d
	picture = HRE_eventPicture
	
	major = yes
	
	is_triggered_only = yes
	
	option = {
		name = ab_wendia.1.a
		add_years_of_income = 1.5
		add_country_modifier = {
			name = wee_charlemagne
			duration = -1
		}
		if = {
			limit = {
				any_country = {
					is_emperor = yes
				}
			}
			emperor = {
				add_stability = -1
				add_imperial_influence = -25
			}
		}
		custom_tooltip = WEE_AACHAEN
	}
}
