	id = ab_gurgan.2
	title = ab_gurgan.2.t
	desc = ab_gurgan.2.d
	picture = BAD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes

		is_year = 1460
		OR = {
			tag = GRG

			mission_completed = grg_temur
