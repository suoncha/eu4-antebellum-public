#Eastern America
country_event = {
	id = colonial_temugids.1
	title = "colonial_temugids.1.t"
	desc = "colonial_temugids.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_eastern_america
			is_core = ROOT
		}
		NOT = { exists = TM1 }
		NOT = {
			has_global_flag = NO_CNS
		}
		OR = {
			is_colonial_nation_of = TMG
			AND = {
				is_colonial_nation_of = CLE
				CLE = { was_tag = TMG }
			}
		}
	}
	option = {
		name = "colonial_temugids.a"
		change_tag = TM1
		swap_free_idea_group = yes
		swap_non_generic_missions = yes
		ab_update_colonial_eastern_america_color = yes
	}
}
