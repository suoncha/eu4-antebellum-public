			tag = SCO

				tag = GBR

			region = british_isles_region

			has_country_flag = kingdom_scotland
		}
	}
	
	fire_only_once = yes 
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5
			has_country_modifier = "the_statue_in_restraint_of_appeals"
		}
		modifier = {
			factor = 0.8
			tag = SCO
			NOT = {
				exists = ABO
				exists = GBR
				exists = GDL
			}
		}
		modifier = {
			factor = 1.5
			tag = SCO
			OR = {
				AND = {
					exists = GBR
					GBR = { 
						was_tag = SCO
					}
				}
				exists = ABO
				exists = GDL
			}
		}
		modifier = {
			factor = 0.6
			was_tag = SCO

				tag = GBR

				limit = {
					religion = catholic
				}
				change_religion = protestant
				add_permanent_province_modifier = {
					name = "religious_zeal_at_conv"
					duration = 9000
				}
			}
		}

			name = "conversion_zeal"
			duration = 3650
		}

			limit = {
				religion = catholic
