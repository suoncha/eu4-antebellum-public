#Eastern America
country_event = {
	id = colonial_yuan.1
	title = "colonial_yuan.1.t"
	desc = "colonial_yuan.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_eastern_america
			is_core = ROOT
		}
		NOT = { exists = YU1 }
		NOT = {
			has_global_flag = NO_CNS
		}
		OR = {
			is_colonial_nation_of = YUA
			AND = {
				is_colonial_nation_of = CLE
				CLE = { was_tag = YUA }
			}
		}
	}
	option = {
		name = "colonial_yuan.a"
		change_tag = YU1
		swap_free_idea_group = yes
		swap_non_generic_missions = yes
		ab_update_colonial_eastern_america_color = yes
	}
}
