	}
}

	fire_only_once = yes 
	
	trigger = {
		OR = {
			tag = CRN

		}
		has_country_flag = crn_mission_unlocked
		has_country_flag = crn_can_fail_mission
		NOT = {
			mission_completed = crn_shatter
		}
		is_at_war = no
		truce_with = ENG
	}
	
	option = {
		name = ab_cornwall.3.a
		clr_country_flag = crn_mission_unlocked
		set_country_flag = crn_mission_failed
		hidden_effect = {
			swap_non_generic_missions = yes
		}
	}
}

	is_triggered_only = yes
	
	major = yes
	
	option = {
		name = ab_cornwall.5.a
		add_prestige = 25
		set_government_rank = 3
		custom_tooltip = ENG_CENTRALIZED
		tooltip = {
			add_province_modifier = {
				name = eng_union_act
				duration = -1
			}
		}
		hidden_effect = {
			british_isles_region = {
				add_province_modifier = {
					name = eng_union_act
					duration = -1
				}
			}
		}
	}
}

#The Birth of the Celtic Union
country_event = {
	id = ab_cornwall.6
	title = ab_cornwall.6.t
	desc = ab_cornwall.6.d
	picture = GOOD_WITH_MONARCH_eventPicture
	
	is_triggered_only = yes
	
	major = yes
	
	option = {
		name = ab_cornwall.6.a
		add_stability = 1
		add_ruler_modifier = {
			name = crn_union
			duration = -1
		}
		france_region = {
			limit = {
				NOT = { is_permanent_claim = ROOT	}
				NOT = { is_core = ROOT	}
			}
			add_permanent_claim = ROOT
		}
		low_countries_region  = {
			limit = {
				NOT = { is_permanent_claim = ROOT	}
				NOT = { is_core = ROOT	}
			}
			add_permanent_claim = ROOT
		}
	}
}
