		}

	id = magyar_events.7
	title = "magyar_events.7.t"
	desc = "magyar_events.7.d"
	picture = CITY_DEVELOPMENT_eventPicture
	
	trigger = {
		OR = {
			tag = MGY
			was_tag = MGY
		}
		owns_core_province = 153
		mission_completed = mgy_conquest_of_the_homeland
		NOT = {
			capital = 153
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	fire_only_once = yes
	
	option = {
		name = magyar_events.7.a
		ai_chance = { factor = 100 }
		add_adm_power = -100
		set_capital = 153
		153 = {
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
		153 = {
			center_of_trade = 1
		}
	}
	
	option = {
		name = magyar_events.7.b
		ai_chance = { factor = 0 }
		capital_scope = {
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
}
