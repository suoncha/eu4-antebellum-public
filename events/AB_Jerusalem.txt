#Latin Empire
country_event = {
	id = ab_jerusalem.5
	title = ab_jerusalem.5.t
	desc = ab_jerusalem.5.d
	picture = AB_BYZFALL_eventPicture
	
	is_triggered_only = yes
	
	major = yes
	
	option = {
		name = ab_jerusalem.5.a
		add_prestige_or_monarch_power = {
			amount = 50
		}
		if = {
			limit = { NOT = { government_rank = 3 } }
			set_government_rank = 3
		}
		add_country_modifier = {
			name = koj_latin_empire
			duration = -1
		}
	}
	
	option = {
		name = ab_jerusalem.5.b
		custom_tooltip = KOJ_LAT_GREECE
		add_country_modifier = {
			name = koj_crown_romans
			duration = -1
		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						area = thrace_area
						area = macedonia_area 
						area = morea_area
						area = aegean_archipelago_area
						area = northern_greece_area
					}
				}
				cede_province = LAT
				add_core = LAT
			}
			create_union = LAT
			LAT = {
				set_government_rank = 3
				set_country_flag = lat_cant_trigger_events
			}
		}
	}
}
