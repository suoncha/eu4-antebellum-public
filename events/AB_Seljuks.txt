# Fate of the Patriarchate
country_event = {
	id = ab_seljuk.8
	title = ab_seljuk.8.t
	desc = ab_seljuk.8.d
	picture = ORTHODOX_eventPicture
	
	
	fire_only_once = yes

	
	trigger ={
		OR = {
			tag = SEL
			was_tag = SEL
			tag = RUM
		}
		capital = 151
		religion_group = muslim
		NOT = { exists = BYZ }
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	
	option = {		# Allow the Patriarchate to exist and keep the current Patriarch
		name = ab_seljuk.8.a
		ai_chance = { factor = 50 }
		capital_scope = {
			add_base_manpower = 3
		}
		every_known_country = {
			limit = {
				religion = orthodox
			}
			add_opinion = {
				who = ROOT
				modifier = allows_the_patriarchate
			}
		}
	}
	option = {		# Close down the Patriarchate and confiscate its properties
		name = ab_seljuk.8.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				any_neighbor_country = {
					religion = orthodox
					army_size = TUR
				}
			}
		}
		add_years_of_income = 0.5
		capital_scope = {
			add_base_tax = 1
		}
		every_known_country = {
			limit = {
				religion = orthodox
			}
			add_opinion = {
				who = ROOT
				modifier = closed_down_the_patriarchate
			}
		}
	}
}
