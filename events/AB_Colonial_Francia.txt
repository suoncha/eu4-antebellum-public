#Louisiana
country_event = {
	id = colonial_francia.1
	title = "colonial_francia.1.t"
	desc = "colonial_francia.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_louisiana
			is_core = ROOT
		}
		NOT = { exists = FR1 }
		NOT = {
			has_global_flag = NO_CNS
		}
		OR = {
			is_colonial_nation_of = FRC
			is_colonial_nation_of = FRK
		}
	}
	option = {
		name = "colonial_francia.a"
		change_tag = FR1
		swap_free_idea_group = yes
		swap_non_generic_missions = yes
		ab_update_colonial_louisiana_color = yes
	}
}
