#Eastern America
country_event = {
	id = colonial_goryeo.1
	title = "colonial_goryeo.1.t"
	desc = "colonial_goryeo.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_eastern_america
			is_core = ROOT
		}
		NOT = { exists = GO1 }
		NOT = {
			has_global_flag = NO_CNS
		}
		OR = {
			is_colonial_nation_of = GOR
			AND = {
				is_colonial_nation_of = CLE
				CLE = { was_tag = GOR }
			}
		}
	}
	option = {
		name = "colonial_goryeo.a"
		change_tag = GO1
		swap_free_idea_group = yes
		swap_non_generic_missions = yes
		ab_update_colonial_eastern_america_color = yes
	}
}
