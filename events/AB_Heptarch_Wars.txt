#Mercia secedes
country_event = {
	id = ab_heptarch_wars.4
	title = ab_heptarch_wars.4.t
	desc = ab_heptarch_wars.4.d
	picture = BATTLE_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		ai = no
		OR = {
			tag = ENG
			tag = GBR
		}
		has_disaster = heptarch_wars
		NOT = {
			exists = MRC
		}
		west_midlands_area = {
			type = all
			owned_by = ROOT
		}
		east_midlands_area = {
			type = all
			owned_by = ROOT
		}
		NOT = {
			dynasty = "Hwicce"
		}
	}
	
	mean_time_to_happen = {
		months = 40
	}
	
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						area = west_midlands_area 
						area = east_midlands_area 
					}
				}
				add_core = MRC
			}
			release = MRC
			MRC = {
				change_religion = PREV
				change_government = monarchy
				add_government_reform = feudalism_reform
				add_legitimacy = 100
				add_manpower = 13
				add_army_tradition = 50
				add_prestige = 30
				add_treasury = 150
				create_general = { tradition = 60 }
				set_dynasty = "Hwicce"
			}
			MRC = {
				add_country_modifier = {
					name = "rebel_heptarch_war"
					duration = 9125
				}
			}
			MRC = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.8
						cavalry = 0.2
					}
				}
				capital_scope = {
					infantry = MRC
					infantry = MRC
					infantry = MRC
					infantry = MRC
					infantry = MRC
					infantry = MRC
					infantry = MRC
					infantry = MRC
				}
				random_owned_province = {
					limit = {
						has_port = yes
					}
					build_to_forcelimit = {
						galley = 0.7
						light_ship = 0.2
						transport = 0.1
					}
				}
			}
		}		
	}
	
	option = {
		name = ab_heptarch_wars.4.a
		ai_chance = { factor = 100 }
		declare_war_with_cb = {
			who = MRC
			casus_belli = cb_core
			war_goal_province = 4372
		}
		if = {
			limit = {
				has_global_flag = rival_supported_heptarch
			}
			custom_tooltip = heptarch_bonus_effect_mercia
			hidden_effect = {
				MRC = {
					capital_scope = {
						infantry = MRC
						infantry = MRC
						infantry = MRC
						infantry = MRC
						infantry = MRC
					}
				}
			}
		}
	}
	
	option = {
		name = ab_heptarch_wars.4.b
		ai_chance = { factor = 0 }
		add_prestige = -25
		every_province = {
			limit = {
				owned_by = MRC
			}
			remove_core = ROOT
		}
	}
	
}
