province_event = {
	id = mnk_flavor.4
	title = mnk_flavor.4.t
	desc = mnk_flavor.4.d
	picture = HAJJ_eventPicture

	hidden = yes
	
	mean_time_to_happen = {
		months = 12
	}

	trigger = {
		owner = {
			has_country_flag = ab_manichean_conversions_flag
		}
		NOT = { religion = owner }
		any_neighbor_province = {
			owned_by = FROM
			religion = owner
		}
	}
	
	option = {
		name = mnk_flavor.4.a
		change_religion = owner
	}
}
