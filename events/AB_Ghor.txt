#Timed Mission Fail
country_event = {
	id = ab_ghor.2
	title = ab_ghor.2.t
	desc = ab_ghor.2.d
	picture = BAD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = GHU
			was_tag = GHU
		}
		has_country_flag = ghu_mission_unlocked
		NOT = {
			mission_completed = ghu_conquest
		}
		NOT = {
			has_country_modifier = ghu_right
		}
	}
	
	immediate = {
		hidden_effect = {
			clr_country_flag = ghu_mission_unlocked
			set_country_flag = ghu_mission_failed
			swap_non_generic_missions = yes
		}
	}
	
	option = {
		name = ab_ghor.2.a
	}
}
