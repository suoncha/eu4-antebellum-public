#Crimea
country_event = {
	id = ab_ghorde.3
	title = ab_ghorde.3.t
	desc = ab_ghorde.3.d
	picture = BATTLE_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		tag = GLH
	}
	
	mean_time_to_happen = {
		months = 25
	}
	
	immediate = {
		set_country_flag = glh_crimea
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = crimean
				}
				add_core = CRI
			}
			release = CRI
			CRI = {
				change_religion = orthodox
				change_government = tribal
				add_government_reform = steppe_horde
				add_horde_unity = 100
				add_manpower = 13
				add_prestige = 30
				add_treasury = 150
				create_general = { tradition = 50 }
			}
			CRI = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.7
						cavalry = 0.3
					}
				}
				random_owned_province = {
					limit = {
						has_port = yes
					}
					build_to_forcelimit = {
						galley = 0.7
						light_ship = 0.2
						transport = 0.1
					}
				}
			}
		}		
	}
	
	option = {
		name = ab_ghorde.3.a
		ai_chance = { factor = 100 }
		set_country_flag = glh_defeat_crimea
		declare_war_with_cb = {
			who = CRI
			casus_belli = cb_core
			war_goal_province = 284
		}
	}
	
	option = {
		name = ab_ghorde.3.b
		ai_chance = { factor = 0 }
		add_prestige = -25
		every_province = {
			limit = {
				owned_by = CRI
			}
			remove_core = GLH
		}
	}
	
}

country_event = {
	id = ab_ghorde.14
	title = ab_ghorde.14.t
	desc = ab_ghorde.14.d
	picture = RELIGIOUS_MIGRANTS_eventPicture
	
	hidden = yes
	
	fire_only_once = yes
	
	trigger = {
		tag = GLH
		ai = yes
		mission_completed = glh_authority
		NOT = {
			mission_completed = glh_branching_1
		}
		NOT = {
			OR = {
				mission_completed = glh_army
				mission_completed = glh_succession
			}
		}
	}
	
	option = {
		name = ab_ghorde.14.a
		if = {
			limit = {
				NOT = {
					mission_completed = glh_army
				}
			}
			complete_mission = glh_army
			remove_country_modifier = gh_army
		}
		if = {
			limit = {
				NOT = {
					mission_completed =  glh_succession
				}
			}
			complete_mission = glh_succession
			remove_country_modifier = gh_court_turbulence
		}
	}
}
