namespace = ab_manic

#Mani Unlock Islam
country_event = {
	id = ab_manic.1
	title = ab_manic.1.t
	desc = ab_manic.1.d
	picture = HAJJ_eventPicture
	
	trigger = {
		religion = manichaean
		NOT = {
			has_country_flag = mani_islam_flag
		}
		OR = {
			OR = {
				num_of_religion = {
					religion = sunni
					value = 0.1
				}
				num_of_religion = {
					religion = shiite
					value = 0.1
				}
				num_of_religion = {
					religion = ibadi
					value = 0.1
				}
			}
			OR = {
				secondary_religion = ibadi
				secondary_religion = sunni
				secondary_religion = shiite
			}
			OR = {
				any_ally = { religion = ibadi }
				any_ally = { religion = shiite }
				any_ally = { religion = sunni }
			}
		}
	}
	
	option = {
		name = ab_manic.1.a
		set_country_flag = mani_islam_flag
	}
}
