#Louisiana
country_event = {
	id = colonial_wendia.1
	title = "colonial_wendia.1.t"
	desc = "colonial_wendia.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_louisiana
			is_core = ROOT
		}
		NOT = { exists = W12 }
		NOT = {
			has_global_flag = NO_CNS
		}
		is_colonial_nation_of = WEE
	}
	option = {
		name = "colonial_wendia.a"
		change_tag = W12
		swap_non_generic_missions = yes
		if = {
			limit = {
				colonial_parent = {
					religion = catholic
				}
			}
			change_religion = catholic
		}
		else_if = {
			limit = {
				colonial_parent = {
					religion = anglican
				}
			}
			change_religion = anglican
		}
		else_if = {
			limit = {
				colonial_parent = {
					religion = reformed
				}
			}
			change_religion = reformed
		}
		else_if = {
			limit = {
				colonial_parent = {
					religion = protestant
				}
			}
			change_religion = protestant
		}
		else_if = {
			limit = {
				colonial_parent = {
					religion = orthodox
				}
			}
			change_religion = orthodox
		}
		else_if = {
			limit = {
				colonial_parent = {
					religion = nestorian
				}
			}
			change_religion = nestorian
		}
	}
}
