defined_text = {
	name = GetPreviewColorAB_LIT
	text = {
		localisation_key = PREVIEW_COLOR
		trigger = {
			can_preview_mission_of_key = {
				key = AB_LIT
			}
			has_preview_mission_flag = yes
		}
	}
	text = {
		localisation_key = empty_text
		trigger = {
			OR = {
				NOT = {
					can_preview_mission_of_key = {
						key = AB_LIT
					}
				}
				NOT = {
					has_preview_mission_flag = yes
				}
			}
		}
	}
}
