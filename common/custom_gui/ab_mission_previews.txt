custom_button = {
    name = ab_select_current_branch_missions
    potential = {
        ai = no #make sure the AI gets an event or something similar where they can choose their branch the moment these buttons get unlocked for the player
        ab_has_branching_missions = yes
        ab_has_preview_mission_flag = yes
    }
    # AB [[[
    trigger = {
        ab_has_selected_a_branch = yes
        ab_can_confirm_branch = yes
    }
    # ]]]
    effect = { ab_select_current_branch = yes }
    tooltip = ab_select_current_branch_missions_title
}

custom_button = {
    name = ab_preview_branch_missions_1
    potential = {
        ai = no
        ab_has_branching_missions = yes

    }
    trigger = { ab_can_select_branch = { value = 1 } }
