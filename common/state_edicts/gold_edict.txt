	potential = {
		always = yes
	}
	
	allow = {
		FROM = {
			any_province_in_state = {
				trade_goods = gold
			}
		}
	}
	
	modifier = {
		local_gold_depletion_chance_modifier = -0.25
	}
	
	color = { 255 234 83 }
	
	ai_will_do = {
		factor = 15
		modifier = {
			factor = 0
			any_province_in_state = {
				NOT = {
					trade_goods = gold
				}
			}
		}
		modifier = {
			factor = 10
			any_province_in_state = {
				trade_goods = gold
			}
		}
	}
}
