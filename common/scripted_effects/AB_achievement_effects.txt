# ========== Etc
ab_achievement_back_page_effect = {
    # First Page
    if = {
        limit = {
            ab_achiev_page = { page = 1 }
        }
        set_variable = {
            which = ab_achievement_page_var
            value = 7
        }
    }
    else = {
        subtract_variable = {            
            which = ab_achievement_page_var
            value = 1
        }
    }
}

ab_achievement_next_page_effect = {
    # Last Page
    if = {
        limit = {
            ab_achiev_page = { page = 7 }
        }
        set_variable = {
            which = ab_achievement_page_var
            value = 1
        }
    }
    else = {
        change_variable = {            
            which = ab_achievement_page_var
            value = 1
        }
    }
}
